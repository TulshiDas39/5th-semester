```
public class Tournament
{
  Performance row = new Performance();

  public Tournament()
  {
    row.Name = "Liverpool";
    row.Score = 15;
  }
  public void DisplayScore()
  {
    string name = row.Name;
    int score = row.Score;
    // ...
  }
}

public class Performance
{
  public string Name
  {
    get;
    set;
  }
  public int Score
  {
    get;
    set;
  }
}
```
