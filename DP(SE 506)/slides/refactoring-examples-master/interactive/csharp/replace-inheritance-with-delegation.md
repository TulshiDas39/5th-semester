public class Engine
{
  //…
  public double Fuel
  { get; set; }
  public double CV
  { get; set; }
}

public class Car: Engine
{
  // ...
  public string Brand
  { get; set; }
  public string Model
  { get; set; }
  public string Name
  {
    get{ return Brand + " " + Model + " (" + CV + "CV)"; }
  }
}

###

```
public class Engine
{
  //…
  public double Fuel
  { get; set; }
  public double CV
  { get; set; }
}

public class Car
{
  // ...
  protected Engine engine;

  public string Brand
  { get; set; }
  public string Model
  { get; set; }
  public string Name
  {
    get{ return Brand + " " + Model + " (" + engine.CV + "CV)"; }
  }

  public Car()
  {
    this.engine = new Engine();
  }
}
```
