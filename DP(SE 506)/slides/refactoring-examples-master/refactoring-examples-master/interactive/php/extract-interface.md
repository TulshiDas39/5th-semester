```
class TimeSheet {
  // ...
  public function charge(Billable $employee, $days) {
    $base = $employee->getRate() * $days;
    if ($employee->hasSpecialSkill()) {
      return $base * 1.05;
    }
    else {
      return $base;
    }
  }
}

interface Billable {
  public function getRate();
  public function hasSpecialSkill();
}

class Employee implements Billable {
  // ...
  public function getRate() {
    // ...
  }
  public function hasSpecialSkill() {
    // ...
  }
}
```
